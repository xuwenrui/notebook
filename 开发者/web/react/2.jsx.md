##### jsx中使用js表达式
###### 1.使用引号传递字符串
###### 2.识别js变量 
###### 3.函数和方法调用
###### 4.使用对象
```jsx
const count = 100;
function App() {
  return (
    <div className="App">
      {/* 1.使用引号传递字符串 */}
      {"this is message"}
      {/* 2.识别js变量  */}
      {count}
      {/* 3.函数和方法调用 */}
      {/* 4.使用对象 */}
      <div style={{ color: "red" }}>thhis is red </div>
    </div>
  );
}
```
