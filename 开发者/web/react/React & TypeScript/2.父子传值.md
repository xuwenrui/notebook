#### 属性传值
```html
<CourseGoal title="Learn React + TS" description="Learn React every day"></CourseGoal>


export default function CourseGoal({
  title,
  description,
}: {
  title: string;
  description: string;
}) {
  return (
    <article>
      <h1>{title}</h1>
      <p>{description}</p>
    </article>
  );
}

```
